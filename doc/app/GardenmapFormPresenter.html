def garden_text_fields_block(gardenYearName, seedPlantSetTerm, options = {})
  markup(:span, class: 'input-block') do |m|
    m << decorated_label(seedPlantSetTerm, "時期", size: 32)
    m << form_builder.select(seedPlantSetTerm, CustomerGardenmap::SEEDPLANTSETTERMS, { include_blank: true }, options)
    m << error_messages_for(seedPlantSetTerm)
    m << decorated_label(gardenYearName, "菜園年度", size: 32)
    m << form_builder.select(gardenYearName, (Time.zone.now.year)..(Time.zone.now.year + 10), { include_blank: false }, options)
    m << error_messages_for(seedPlantSetTerm)
    m << text_field(:vegetable_id, options)
    m << error_messages_for(:vegetable_id)
    m << decorated_label(:main_vegetable_name, "野菜名", size: 32)
    m << text_field(:main_vegetable_name, options)
    m << error_messages_for(:main_vegetable_name)
  end
end
